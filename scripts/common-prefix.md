Get common [prefix] of strings.

```javascript
const commonPrefix = require('@extra-string/common-prefix');
// commonPrefix(<string-1>, <string-2>)

commonPrefix('peacock', 'peahen');
// 'pea'
commonPrefix('inception', 'interstellar');
// 'in'
commonPrefix('mars', 'tars');
// ''
```
