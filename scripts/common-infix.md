Get common [infix] between strings.

```javascript
const commonInfix = require('@extra-string/common-infix');
// commonInfix(<string-1>, <string-2>)

commonInfix('mangala', 'mangalyaan');
// 'mangal'
commonInfix('easter', 'tertiary');
// 'ter'
commonInfix('dismiss', 'mississipi');
// 'miss'
```
